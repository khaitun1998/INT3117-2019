```sh
1   	static int largestContiguousSum(int arr) {
2		int i,len=arr.length,cursum=0,maxsum=Integer.MIN_VALUE;
3		if(len==0)	//empty array
4		for(i=0;i<len;i++){
5			cursum+=arr[i];
6			if(cursum>maxsum){
7				maxsum=cursum;
8			}
9			if(cursum<=0){
10				cursum=0;
11			}
12		}
13		return maxsum;
14	}
```
